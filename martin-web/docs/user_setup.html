var checkTimingFlag = function (textInput) {
    if (textInput.indexOf(' -t') > -1) {
        wantTimingInformation = true;
        textInput = textInput.replace(' -t', '');
    } else {
        wantTimingInformation = false;
    }
    return textInput;
}
